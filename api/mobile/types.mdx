## Overview

Complete TypeScript type reference for `@navai/voice-mobile`.

## Hook Types

### UseMobileVoiceAgentOptions

Options for `useMobileVoiceAgent` hook.

```tsx
type UseMobileVoiceAgentOptions = {
  runtime: ResolveNavaiMobileApplicationRuntimeConfigResult | null;
  runtimeLoading: boolean;
  runtimeError: string | null;
  navigate: (path: string) => void;
};
```

<ParamField path="runtime" type="ResolveNavaiMobileApplicationRuntimeConfigResult | null" required>
  Runtime configuration or `null` if not loaded
</ParamField>

<ParamField path="runtimeLoading" type="boolean" required>
  Whether runtime is loading
</ParamField>

<ParamField path="runtimeError" type="string | null" required>
  Runtime loading error
</ParamField>

<ParamField path="navigate" type="(path: string) => void" required>
  Navigation function
</ParamField>

### UseMobileVoiceAgentResult

Return type of `useMobileVoiceAgent` hook.

```tsx
type UseMobileVoiceAgentResult = {
  status: 'idle' | 'connecting' | 'connected' | 'error';
  error: string | null;
  isConnecting: boolean;
  isConnected: boolean;
  start: () => Promise<void>;
  stop: () => Promise<void>;
};
```

<ResponseField name="status" type="'idle' | 'connecting' | 'connected' | 'error'">
  Current session status
</ResponseField>

<ResponseField name="error" type="string | null">
  Error message if status is error
</ResponseField>

<ResponseField name="isConnecting" type="boolean">
  Convenience flag for connecting state
</ResponseField>

<ResponseField name="isConnected" type="boolean">
  Convenience flag for connected state
</ResponseField>

<ResponseField name="start" type="() => Promise<void>">
  Start voice session
</ResponseField>

<ResponseField name="stop" type="() => Promise<void>">
  Stop voice session
</ResponseField>
